## Transcribe

End-to-end transcription — from upload to result in one call.

<Tip>
  Pass in a local file, binary data, or a remote URL — and let the method handle the rest.
</Tip>

<CodeGroup>

```python Python
from gladiaio_sdk import GladiaClient

gladia_client = GladiaClient(api_key="YOUR_GLADIA_API_KEY").prerecorded()

transcription = gladia_client.transcribe(
    "YOUR_AUDIO_URL_OR_LOCAL_PATH",
    {
        "pii_redaction": True,
        "pii_redaction_config": {
            "entity_types": ["GDPR"],
            "processed_text_type": "MASK",
        },
        "language_config": {
            "languages": ["en", "ko", "zh", "mn", "ru", "ja"],
            "code_switching": True,
        },
        "custom_vocabulary_config": {
            "vocabulary": ["aaruul", "mutton"],
            "default_intensity": 0.6,
        },
    },
)

print(transcription.result.transcription.full_transcript)
```

```javascript JavaScript
import { GladiaClient } from '@gladiaio/sdk';

const gladiaClient = new GladiaClient({ apiKey: 'YOUR_GLADIA_API_KEY' });

const transcription = await gladiaClient.preRecorded().transcribe(
  "YOUR_AUDIO_URL_OR_LOCAL_PATH",
  {
    pii_redaction: true,
    pii_redaction_config: {
      entity_types: ["GDPR"],
      processed_text_type: "MASK",
    },
    language_config: {
      languages: ["en", "ko", "zh", "mn", "ru", "ja"],
      code_switching: true,
    },
    custom_vocabulary_config: {
      vocabulary: ["aaruul", "mutton"],
      default_intensity: 0.6,
    },
  }
);

console.log(transcription.result?.transcription?.full_transcript ?? "");
```

</CodeGroup>

## Individual steps

The building blocks behind `transcribe()` — call them individually when you need finer control over the flow.

### Upload your audio

Upload a local file and pass the returned `audio_url` to the next step.

<CodeGroup>

```python Python
from gladiaio_sdk import GladiaClient

gladia_client = GladiaClient(api_key="YOUR_GLADIA_API_KEY").prerecorded()

upload_response = gladia_client.upload_file("/path/to/your/audio/your_audio_file.mp3")

print(upload_response.audio_url)
print(upload_response.audio_metadata.id)
```

```javascript JavaScript
import { GladiaClient } from '@gladiaio/sdk';

const gladiaClient = new GladiaClient({ apiKey: 'YOUR_GLADIA_API_KEY' });

const uploadResponse = await gladiaClient.preRecorded().uploadFile("/path/to/your/audio/your_audio_file.mp3");

console.log(uploadResponse.audio_url);
console.log(uploadResponse.audio_metadata.id);
```

```bash cURL
curl --request POST \
  --url https://api.gladia.io/v2/upload \
  --header 'Content-Type: multipart/form-data' \
  --header 'x-gladia-key: YOUR_GLADIA_API_KEY' \
  --form audio=@/path/to/your/audio/your_audio_file.mp3
```

</CodeGroup>

### Create a transcription job

Pass the `audio_url` from the previous step along with your transcription options.

<CodeGroup>

```python Python
from gladiaio_sdk import GladiaClient

gladia_client = GladiaClient(api_key="YOUR_GLADIA_API_KEY").prerecorded()

job = gladia_client.create({
    "audio_url": "YOUR_AUDIO_URL",
    "language_config": {
        "languages": [],
        "code_switching": False,
    },
    "diarization": True,
    "diarization_config": {
        "number_of_speakers": 3,
        "min_speakers": 1,
        "max_speakers": 5,
    },
    "translation": True,
    "translation_config": {
        "model": "base",
        "target_languages": ["fr", "en"],
        "context_adaptation": True,
        "context": "Business meeting discussing quarterly results",
        "informal": False,
    },
    "subtitles": True,
    "subtitles_config": {
        "formats": ["srt", "vtt"],
    },
})

```

```javascript JavaScript
import { GladiaClient } from '@gladiaio/sdk';

const gladiaClient = new GladiaClient({ apiKey: 'YOUR_GLADIA_API_KEY' });

const job = await gladiaClient.preRecorded().create({
  audio_url: "YOUR_AUDIO_URL",
  language_config: {
    languages: [],
    code_switching: false,
  },
  diarization: true,
  diarization_config: {
    number_of_speakers: 3,
    min_speakers: 1,
    max_speakers: 5,
  },
  translation: true,
  translation_config: {
    model: "base",
    target_languages: ["fr", "en"],
    context_adaptation: true,
    context: "Business meeting discussing quarterly results",
    informal: false,
  },
  subtitles: true,
  subtitles_config: {
    formats: ["srt", "vtt"],
  },
});

console.log(job.id);
console.log(job.result_url);
```

```bash cURL
curl --request POST \
  --url https://api.gladia.io/v2/pre-recorded \
  --header 'Content-Type: application/json' \
  --header 'x-gladia-key: YOUR_GLADIA_API_KEY' \
  --data '{
  "audio_url": "YOUR_AUDIO_URL",
  "language_config": {
    "languages": [],
    "code_switching": false
  },
  "diarization": true,
  "diarization_config": {
    "number_of_speakers": 3,
    "min_speakers": 1,
    "max_speakers": 5
  },
  "translation": true,
  "translation_config": {
    "model": "base",
    "target_languages": ["fr", "en"],
    "context_adaptation": true,
    "context": "Business meeting discussing quarterly results",
    "informal": false
  },
  "subtitles": true,
  "subtitles_config": {
    "formats": ["srt", "vtt"]
  }
}'
```
